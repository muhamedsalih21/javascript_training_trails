    override func viewDidLoad() {
        super.viewDidLoad()
        //Circle Points
        let center = CGPoint (x: myCircle.frame.size.width / 2, y: myCircle.frame.size.height / 2)
        let circleRadius = myCircle.frame.size.width / 2
        let circlePath = UIBezierPath(arcCenter: center, radius: circleRadius, startAngle: CGFloat(M_PI), endAngle: CGFloat(M_PI * 2), clockwise: true)
        
        
        semiCircleLayer.path = circlePath.cgPath
        semiCircleLayer.strokeColor = UIColor.red.cgColor
        semiCircleLayer.fillColor = UIColor.blue.cgColor
        semiCircleLayer.lineWidth = 8
        semiCircleLayer.strokeStart = 0
        semiCircleLayer.strokeEnd  = 1
        myCircle.layer.addSublayer(semiCircleLayer)
    }

    override func didReceiveMemoryWarning() {

        // Dispose of any resources that can be recreated.
    }

import UIKit

class ViewController: UIViewController {

    internal var c :Int = 0
    @IBOutlet var time: UILabel!
    @IBAction func start(_ sender: Any) {
        timer = Timer.scheduledTimer(timeInterval: 1.0,target:self,selector:#selector(updateCountDown),userInfo:nil,repeats: true)
    }
    @objc func updateCountDown()
    {
        c+=1
        time.text=String(format: "%02d:%02d:%02d",c/3600,(c%3600)/60,(c%3600)%60)
    }
    
    @IBAction func stop(_ sender: Any) {
        timer.invalidate()
    }
    
    
    @IBAction func reset(_ sender: Any) {
        c=0
        stop(sender)
        time.text!="00:00:00"
    }
    var timer=Timer()
    
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view, typically from a nib.
    }

    override func didReceiveMemoryWarning() {
        super.didReceiveMemoryWarning()
        // Dispose of any resources that can be recreated.
    }


}

import UIKit
class ViewController: UIViewController {
@IBOutlet var status: UILabel!
@IBOutlet var dept: UITextField!
@IBOutlet var name: UITextField!
var databasePath = String()
override func viewDidLoad() {
super.viewDidLoad()
// Do any additional setup after loading the view, typically from a nib.
let filemgr = FileManager.default
let dirPaths = filemgr.urls(for: .documentDirectory,
in: .userDomainMask)

databasePath = dirPaths[0].appendingPathComponent(&quot;contacts.db&quot;).path

if !filemgr.fileExists(atPath: databasePath as String) {

let contactDB = FMDatabase(path: databasePath as String)

if contactDB == nil {
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
}

if (contactDB.open()) {
let sql_stmt = &quot;CREATE TABLE IF NOT EXISTS CONTACTS (ID INTEGER PRIMARY
KEY AUTOINCREMENT, NAME TEXT, DEPT TEXT)&quot;
if !(contactDB.executeStatements(sql_stmt)) {
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
}
contactDB.close()
} else {
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
}
}

}
@IBAction func FindData(_ sender: Any) {
let contactDB = FMDatabase(path: databasePath as String)

if (contactDB.open()) {
let querySQL = &quot;SELECT dept FROM CONTACTS WHERE name = &#39;\(name.text!)&#39;&quot;

let results:FMResultSet? = contactDB.executeQuery(querySQL, withArgumentsIn: [])


if results?.next() == true {
dept.text = results?.string(forColumn: &quot;dept&quot;)
status.text = &quot;Record Found&quot;
} else {
status.text = &quot;Record not found&quot;
dept.text = &quot;&quot;
}
contactDB.close()
} else {
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
}

}
@IBAction func SaveData(_ sender: Any) {

let contactDB = FMDatabase(path: databasePath as String)

if (contactDB.open()) {

let insertSQL = &quot;INSERT INTO CONTACTS (name, dept) VALUES (&#39;\(name.text!)&#39;,
&#39;\(dept.text!)&#39;)&quot;

let result = contactDB.executeUpdate(insertSQL,
withArgumentsIn: [])

if !result {
status.text = &quot;Failed to add contact&quot;
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
} else {
status.text = &quot;Contact Added&quot;
name.text = &quot;&quot;
dept.text = &quot;&quot;
}
} else {
print(&quot;Error: \(contactDB.lastErrorMessage())&quot;)
}
}
override func didReceiveMemoryWarning() {
super.didReceiveMemoryWarning()
// Dispose of any resources that can be recreated.
}

}
